/*Esta funcion permite pasar un arreglo de caracter o un string,

string cadana_Array(string cadena){
    int aux[20]={};
    int l=0,i=0;
    string lectura;
    stringstream input_stringstream(cadena);   // Convierte la cadena a un stream
    while (getline(input_stringstream, lectura, ',')||getline(input_stringstream, lectura, '\0')) //tomar los valores que estan antes de la coma o el salto de linea

    {
        l=atoi(lectura.c_str()); // funcion que coge cada caracter del arreglo y lo pasa a un numero entero
        aux[i]=l;
        cout << "Un valor: " << lectura << endl;
        cout << "Un aux[i]: " << *aux << endl;
        i++;
    }
    return 0;
};

void loop()
{
  if(Serial.available()>0){ // si detecta algo en serial

int convertir_binario(int NUM){
